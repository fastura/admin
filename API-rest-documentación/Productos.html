REGISTRAR PRODUCTOS​

Para registrar productos debe enviar hacia:

  • Url: http://demo.fastura.app/api/item

Ejemplo:

  1. JSON A ENVIAR
{
    "item_type_id": "01",
    "internal_id": "123456",
    "item_code": null,
    "item_code_gs1": null,
    "description": "el nombre",
    "name": "la descripción",
    "second_name": null,
    "unit_type_id": "NIU",
    "currency_type_id": "PEN",
    "sale_unit_price": "5",
    "purchase_unit_price": 0,
    "has_isc": false,
    "system_isc_type_id": null,
    "percentage_isc": 0,
    "suggested_price": 0,
    "sale_affectation_igv_type_id": "10",
    "purchase_affectation_igv_type_id": "10",
    "calculate_quantity": false,
    "stock": 0,
    "stock_min": 0,
    "has_igv": true,
    "has_perception": false,
    "item_unit_types": [],
    "percentage_of_profit": 0,
    "percentage_perception": 0,
    "is_set": false,
    "account_id": null,
    "category_id": null,
    "brand_id": null,
    "date_of_due": null,
    "lot_code": null,
    "lots_enabled": false,
    "lots": []
}
```
